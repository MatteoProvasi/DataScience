stop = stopwords.words('english')

# parole aggiuntive
new_stop = ["amp", "http", "would", "one", "could", "still", "via", "two", "also", "im", "must", "its", "already",
            "actually", "really", "well", "yet", "always", "since", "absolutely", "youre", "clearely", "theres", "ive", 
            "lets", "shes", "yeah", "rt", "co", "go", "nh", "le", "were", "oh", "gt"]

# parole che non devono essere rimosse
no_stop = ["and", "but", "don't", "doesn't", "didn't", "hadn't", "hasn't", "haven't", "isn't", "mustn't",
           "needn't", "couldn't", "ain", "aren't", "shouldn't", "wasn't", "weren't", "won't", "wouldn't"]

stop = stop + new_stop
for token in no_stop:
    stop.remove(token)
