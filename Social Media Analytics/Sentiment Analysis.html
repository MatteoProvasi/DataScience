# numero di tweet ogni 5 minuti
lens = c()
for (i in 1:(length(t_seq)-1)) {
  dat = tweets[which(tweets$Data > t_seq[i] & tweets$Data < t_seq[i+1]), ]
  lens = append(lens, nrow(dat))
}


# numero cumulativo di tweets ogni 5 minuti
lens2 = c(1, lens[1])
for (i in 1:(length(lens)-1)) {
  lens2 = append(lens2, lens2[i+1] + lens[i+1])
}
lens2 = append(lens2, nrow(tweets))

# visualizzazione ad intervalli di 5 minuti
start_time = Sys.time()
for (i in 1:(length(lens2)-1)) {
  
  # 10 emozioni secondo il metodo "nrc"
  result = get_nrc_sentiment(as.character(tweets$Tweets)[lens2[i]:lens2[i+1]])
  result1 = data.frame(t(result)) 
  
  new_result = data.frame(rowSums(result1)) # somma per riga
  names(new_result)[1] = "count"
  new_result = cbind("sentiment"=rownames(new_result), new_result)
  rownames(new_result) = NULL
  
  table = rbind(table, as.integer(t(new_result)[2, ]))   # emotions salvate in un dataframe
  
  
  # sentiment
  word.df = as.vector(t(tweets$Tweets)[lens2[i]:lens2[i+1]])
  
  sent.value_afinn = get_sentiment(word.df, method = "afinn")
  sent.value_bing = get_sentiment(word.df, method = "bing")
  sent.value_nrc = get_sentiment(word.df, method = "nrc")
  sent.value_syuzhet = get_sentiment(word.df, method = "syuzhet")
  
  sentiment_afinn = append(sentiment_afinn, mean(sent.value_afinn))
  sentiment_bing = append(sentiment_bing, mean(sent.value_bing))
  sentiment_nrc = append(sentiment_nrc, mean(sent.value_nrc))
  sentiment_syuzhet = append(sentiment_syuzhet, mean(sent.value_syuzhet))
 
  
  # plot 8 sentiment
  sentiment1 = qplot(sentiment, data=new_result[1:8, ], weight=count, geom="bar", fill=sentiment) +
    ggtitle(paste0("Brexit Emotions between", 
                   substr(t_seq[i], start=11, stop=stri_length(t_seq[i])), " and", 
                   substr(t_seq[i+1], start=11, stop=stri_length(t_seq[i+1]))))
  
  # plot neg/pos
  sentiment2 = qplot(sentiment, data=new_result[9:10, ], weight=count/sum(count), geom="bar", fill=sentiment) + 
      ggtitle(paste0("Brexit positive/negative tweets between", 
                     substr(t_seq[i], start=11, stop=stri_length(t_seq[i])), " and", 
                     substr(t_seq[i+1], start=11, stop=stri_length(t_seq[i+1]))))
  
  #grid.arrange(sentiment1, sentiment2, ncol=2) # visualizzazione
  
  eight_sentiment[[i]] = sentiment1
  posneg_sentiment[[i]] = sentiment2
  
  
  print(paste0("Progression: ", as.character(round(100*(i/length(lens2)), 3)), "%"))
  print(Sys.time() - start_time)
}

Il loop permette la creazione di un grafico a barre che rappresenta landamento di ogni emozione.

# plot dei sentiment sul tempo
plot_list = list()
i = 1
for (colname in colnames(table3)[1:10]){
  graph = ggplot(data=table3, aes_string(y=colname, x="Time", fill=colname)) +                 # dati
    scale_fill_gradient(low = hue_pal()(10)[i], high = hue_pal()(10)[i]) +                     # palette
    geom_bar(stat="identity") + 
    scale_y_continuous(limits=c(min(table3[, i])*0.95, max(table3[, i])*1), oob=rescale_none)
  
  plot_list[[i]] = graph
  i = i + 1
}
